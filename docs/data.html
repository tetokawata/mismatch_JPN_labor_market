3.2.3 Merge

df.pref <-
  raw.full %>%
  rbind(raw.part) %>%
  spread(type,n) %>%
  mutate(p = Hir/Seeker,
         q = Hir/Vacancy,
         t = Vacancy/Seeker,
         date = make_date(year, month)
         ) %>%
  mutate(large.pref = if_else(pref.id == 1,
                              "Hokkaido",
                              "A"),
         large.pref = if_else(pref.id >= 2 &
                                pref.id <= 7,
                              "Tohoku",
                              large.pref),
         large.pref = if_else(pref.id >= 8 &
                                pref.id <= 10,
                              "Kita Kanto & Koushin",
                              large.pref),
         large.pref = if_else(pref.id >= 11 &
                                pref.id <= 14,
                              "Minami Kanto",
                              large.pref),
         large.pref = if_else(pref.id >= 15 &
                                pref.id <= 18,
                              "Hokuriku",
                              large.pref),
         large.pref = if_else(pref.id >= 19 &
                                pref.id <= 20,
                              "Kita Kanto & Koushin",
                              large.pref),
         large.pref = if_else(pref.id >= 21 &
                                pref.id <= 24,
                              "Tokai",
                              large.pref),
         large.pref = if_else(pref.id >= 25 &
                                pref.id <= 30,
                              "Kinki",
                              large.pref),
         large.pref = if_else(pref.id >= 31 &
                                pref.id <= 35,
                              "Chugoku",
                              large.pref),
         large.pref = if_else(pref.id >= 36 &
                                pref.id <= 39,
                              "Shikoku",
                              large.pref),
         large.pref = if_else(pref.id >= 40 &
                                pref.id <= 46,
                              "Kyusyu",
                              large.pref),
         large.pref = if_else(pref.id == 47,
                              "Okinawa",
                              large.pref),
         large.pref = factor(large.pref,
                             levels = c("Hokkaido",
                                        "Tohoku",
                                        "Kita Kanto & Koushin",
                                        "Minami Kanto",
                                        "Hokuriku",
                                        "Tokai",
                                        "Kinki",
                                        "Chugoku",
                                        "Shikoku",
                                        "Kyusyu",
                                        "Okinawa"))
         )
